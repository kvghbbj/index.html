	<script>
		// Fullscreen button functionality
        document.getElementById('fullscreen-button').addEventListener('click', function() {
            var elem = document.getElementById('openfl-content');
            if (elem.requestFullscreen) {
                elem.requestFullscreen();
            } else if (elem.mozRequestFullScreen) { // Firefox
                elem.mozRequestFullScreen();
            } else if (elem.webkitRequestFullscreen) { // Chrome, Safari and Opera
                elem.webkitRequestFullscreen();
            } else if (elem.msRequestFullscreen) { // IE/Edge
                elem.msRequestFullscreen();
            }
        });

        // Ensure 'lime' is defined before calling 'lime.embed'
        if (typeof lime !== 'undefined') {
            lime.embed("Level Devil", "openfl-content", 854, 480, { parameters: {} });
        } else {
            console.error("Lime is not defined. Please include the lime.js script.");
        }
